@maskedAccountNumber = @{

@formatDate(yyyymmdd: String) = @{
  val year = yyyymmdd.slice(2,4)
  val month = yyyymmdd.slice(4,6)
  val day = yyyymmdd.slice(6,8)
  s"$day/$month/$year"
}
